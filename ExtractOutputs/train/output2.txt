            var cache = {};
            var instance = this;
            this.isListening = function (topicName, subscriberName) {
                var ret = false;
                if (cache[topicName]) {

                    }).length > 0;
                }
                return ret;

            this.listen = function (topicName, subscriberName, subscribingFunction) {
                if (cache[topicName] == undefined) {
                    cache[topicName] = [];

                    return;
